Projede arduinoya bağlı olan elektronik cihazlar arduino içerisine gömdüğümüz program tarafından kontrol edilmektedir. Arduino içerisindeki program ise, Nodemcu'ya bağlı olan 4 pinden okunan digital değerler ile kontrol edilmektedir. Bu pinlerden biri robotun hareket edip etmemesini kontrol eder. Diğer 3 pinimiz 2^n = 2^3 = 8 hareketi kontrol etmeyi sağlar. Örneğin bu pinlerden sırayla 1-0-0 (HIGH-LOW-LOW) okunduğunda ve hareket pininden HIGH okunduğunda robot Z ekseninde + yönde hareket ettirir.

Programda ilk olarak motor pinleri, kontrol giriş pinleri ve bekleme süresi global sabitler olarak tanımlanmıştır. Servo motor için Servo.h kütüphanesi eklenmiş ve ilk değer belirlenmiştir. 

#include <Servo.h>

const int motor1[] = {4, 5, 6, 7}; // Step Motor 1 - X ekseni 
const int motor2[] = {8, 9, 10, 11}; // Step Motor 2 - Z ekseni 
const int motor3[] = {14, 15, 16, 17}; // Step Motor 3 - Y ekseni
        
const int delayTime = 3;
const int controlPins[] = {49, 47, 45};
const int actionPin = 51;
        
const int servoPin = 13; // Servo Motor Pini
const Servo servoMotor;
int servoValue = 0;

Setup fonksiyonu program başlatılırken yalnızca bir kez çalıştırılan kurulum fonksiyonudur. Burada pinlerin modları belirlenmiş, servo motor, pinine bağlanıp başlangıç açısı verilmiş ve seri monitör başlatılmıştır. 

void setup() {
    
    /* For döngüsü içerisinde, başlangıçta tanımlanan motor pinleri
    OUTPUT moduna getirilir. */

    for(int i=0; i<4; i++) {
      pinMode(motor1[i], OUTPUT);
      pinMode(motor2[i], OUTPUT);
      pinMode(motor3[i], OUTPUT);
      }

    /* Control pinleri ise INPUT moduna getirilir. */

    for(int i=0; i<3; i++) {
      pinMode(controlPins[i], INPUT);
      }

      /*  Hareket pini INPUT yapılır. */

      pinMode(actionPin, INPUT);
      
      /* Servo kütüphanesinin Servo sınıfından örneklenen servoMotor
      değişkeni ile servo motorun bağlı olduğu pin iliştirilir. */
        
      servoMotor.attach(servoPin);

      /* Ve servo motora başlangıç olarak servoValue'nin değeri verilir. */
        
      servoMotor.write(servoValue);
      
      Serial.begin(9600);
  }

Step motor arduinoya, step motor sürücü kartı ile bağlanmıştır. Bu kart üzerinde IN1, IN2, IN3, IN4 olarak berlirtilen giriş pinleri, programımızda tanımladığımız step motor çıkış pinlerine bağlıdırlar. Arduinonun bu çıkış pinlerine, sırasıyla 5V verilerek, step motorun bir tam adım atması sağlanır.

Step motorların ileri ve geri adımları için iki ayrı fonksiyon yazılmıştır. Bu fonksiyonlar doğru pinlere doğru sırada sinyaller vererek hareketi sağlar. Fonksiyonlar parametre olarak hareket ettirecekleri motor pinlerini içeren integer diziyi alırlar. Ve geriye değer döndürmeden motorun her pinine sırasıyla HIGH değeri vererek bir tam adım atmasını sağlarlar. 

void moveNext(int motor[]) {
  
  int _step = 0;

  /* Burada tanımlanan _step değişkeni, motorun
  hangi indexteki pinine HIGH verileceğini tutmaktadır.
  Aynı zamanda bir tam adımın kaçıncı çeyrek adımında olduğunu temsil etmektedir.
  Bu fonksiyonda ileri tam adım atıldığı için 0. indexteki pinden adım başlamıştır. */
  
  while(_step < 4) {
  
    /* 4 pinin her birine sırasıyla HIGH ve kalanlara LOW verilmesi için
    için 4 kez işleme giren while döngüsü yazılmıştır. */
     
    for(int i=0; i<4; i++) {
    
    /* Ve for döngüsü içerisinde motorun _step indexindeki pinine
    HIGH diğer pinlerine LOW verilmiştir. */
  
    if(i != _step){
        digitalWrite(motor[i], LOW);
      }else {
          digitalWrite(motor[i], HIGH);
        }
    }

    delay(delayTime);

    /* Her pine değeri verildikten sonra delayTime kadar motorun 
    hareket edebilmesi için beklenmiştir.
    Ardından _step değişkeni artırılarak sonraki çeyrek adıma geçilmiştir. */
    
    _step ++;
   }
}
                    
void moveBack(int motor[]) {
  
  /* Bu fonksiyonun moveNext fonksiyonundan tek farkı, tam adıma, 
  motorun 3. indexteki pinine HIGH vererek başlaması ve
  _step değişkeninin azaltarak ters yönde adım atılmasını sağlamasıdır. */

  int _step = 3;

  while(_step > -1) {

   for(int i=0; i<4; i++) {
    
    if(i != _step){
        digitalWrite(motor[i], LOW);
      }else {
          digitalWrite(motor[i], HIGH);
        }
    }
    delay(delayTime);
    _step --;
   }
}

Gripperın hareketini sağlamak için servo motoru kontrol eden bir fonksiyon yazılmıştır. Kontrol pinlerinden 000 okunduğunda servoValue azaltılır, 111 okunduğunda ise artırılır. Ve bu fonksiyon çağırılarak gripper açılıp kapanır. 

void gripper() {

    /* Bu fonksiyon servo motor aracılığı ile 
    gripperı hareket ettirmek için kullanılır. */
    
    /* */
    servoValue = servoValue > 160 ? 160: (servoValue < 40 ? 40: servoValue);
    Serial.println("\nServo value is ");
    Serial.print(servoValue);
    Serial.println("");
     
    /* Servo motor için hazırlanan bu değişkenin değeri 
    write metodu ile servo motora verilir */   
    servoMotor.write(servoValue);   
    delay(10); 
    
   }

Programın ana döngüsü olan loop fonksiyonu içerisinde robot kolun kontrolü sağlanmıştır. Bu fonksiyonda joystick pinlerinden değerler okunur, ve o değerlere göre moveBack, moveNext, gripper fonksiyonları çağırılır. 

void loop() {

    /* Hangi motorun hangi yönde döneceğini belirleyecek olan değer */  

    int sw = 0;

    
    /* Bu değer controlPins'den okunan okunan digital değerlerin sırayla 
    2'nin 0., 1. ve 2. kuvvetleri ile çarpılıp toplanmasıyla bulunur. 
    Bu pinler 3 step motorun ve servo motorun iki yönde dönüşlerinden 
    hangisinin gerçekleşeceğini belirler. Yani 8 durum 3 pin ile kontrol edilir. */ 

    for(int i=0; i<3; i++) {

        sw += ( ( digitalRead(controlPins[i])==HIGH ) ? 1 : 0 ) * (i==0 ? 1 : (i==1 ? 2: 4));
    }

    if(digitalRead(actionPin) == HIGH) {
        
        Serial.print("\nSwitch value: ");
        Serial.println(sw);
        Serial.println("Action pin is HIGH");

        switch(sw) {
          /// y degeri 470'den küçükse
          case 0:
            servoValue += 5;
            gripper();
            break;
          case 1:
            moveNext(motor1);
            break;
          case 2:
            moveBack(motor1);
            break;
          case 3:
            moveNext(motor2);
            break;
          case 4:
            moveBack(motor2);
            break;
          case 5:
            moveNext(motor3);
            break;
          case 6:
            moveBack(motor3);
            break;
          case 7:
            servoValue -= 5;
            gripper();
            break;
        }
        delay(2);
    }
}
